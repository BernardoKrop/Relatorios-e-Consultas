 SELECT Solicitacao."DocNum", 
	   Solicitacao."DocEntry", 
	   Solicitacao."DocDate",    
	   Solicitacao."CardName", 
	   Solicitacao."CardCode", 
	   Solicitacao."Address",
       SolItem."ItemCode", 
       SolItem."Dscription",
       SUM(SolItem."Quantity")     AS "Quantity",    
       MetodoPagto."PymntGroup",  
       SolItem."unitMsr",  
       Solicitacao."Comments", 
       NCM."NcmCode", 
       Solicitacao."ReqName"       AS "Solicitante", 
       Filial."BPLName"            AS "EMP Nome", 
       Filial."AddrType"           AS "EMP tipo", 
       Filial."Street"             AS "EMP Rua", 
       Filial."StreetNo"           AS "EMP num", 
       Filial."Building"           AS "EMP Comp", 
       Filial."ZipCode"            AS "EMP CEP", 
       Filial."Block"              AS "EMP Bairro", 
       Filial."City"               AS "EMP Cidade", 
       Filial."State"              AS "EMP Estado", 
       (SELECT "Phone1" 
          FROM "{?Schema@}".OADM) AS "EMP tel", 
       (SELECT "Phone2" 
          FROM "{?Schema@}".OADM) AS "EMP DDD", 
       (SELECT "E_Mail" 
          FROM "{?Schema@}".OADM) AS "EMP Email", 
       Filial."TaxIdNum"           AS "EMP CNPJ", 
       Filial."TaxIdNum2"          AS "EMP IE",
       SolItem."FreeTxt",
       Solicitacao."U_LocalEntrega"  AS "Local Entrega"
FROM "{?Schema@}".OPRQ            Solicitacao
    INNER JOIN "{?Schema@}".PRQ1  SolItem      ON Solicitacao."DocEntry"  = SolItem."DocEntry" 
    LEFT  JOIN "{?Schema@}".OBPL  Filial       ON Filial."BPLId"          = Solicitacao."BPLId" 
    LEFT  JOIN "{?Schema@}".OCTG  MetodoPagto  ON Solicitacao."GroupNum"  = MetodoPagto."GroupNum" 
    LEFT  JOIN "{?Schema@}".OITM  Produto      ON Produto."ItemCode"      = SolItem."ItemCode" 
    LEFT  JOIN "{?Schema@}".ONCM  NCM          ON Produto."NCMCode"       = NCM."AbsEntry"
GROUP BY Solicitacao."DocNum", 
	   Solicitacao."DocEntry", 
	   Solicitacao."DocDate",    
	   Solicitacao."CardName", 
	   Solicitacao."CardCode", 
	   Solicitacao."Address",
       SolItem."ItemCode", 
       SolItem."Dscription",
       SolItem."Quantity",    
       MetodoPagto."PymntGroup",  
       SolItem."unitMsr",  
       Solicitacao."Comments", 
       NCM."NcmCode", 
       Solicitacao."ReqName", 
       Filial."BPLName", 
       Filial."AddrType", 
       Filial."Street", 
       Filial."StreetNo", 
       Filial."Building", 
       Filial."ZipCode", 
       Filial."Block", 
       Filial."City", 
       Filial."State", 
       Filial."TaxIdNum", 
       Filial."TaxIdNum2",
       SolItem."FreeTxt",
       Solicitacao."U_LocalEntrega"
