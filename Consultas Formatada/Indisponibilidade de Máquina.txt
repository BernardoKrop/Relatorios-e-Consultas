SELECT Ordem."NrOS"                 AS "Nr Ordem",
       INITCAP(Fazenda."Name")      AS "Fazenda",
       INITCAP(Motivo."Name")       AS "Motivo",
       INITCAP(Classe."Name")       AS "Classe",
       Ordem."Status",
	   Ordem."Avan√ßo", 	 
	   Ordem."IdEquipamento"        AS "Frota",
	   INITCAP(Equip."U_DscModelo") AS "Equipamento",
       Ordem."TmpDecorrido"         AS "Tempo Decorrido (Hrs)",
       Ordem."TmpNumero"            AS "Tempo Decorrido (Nr)", 
       Ordem."DataEntrada",
       MONTH(Ordem."DataEntrada")   AS "Mes Entrada",
       YEAR(Ordem."DataEntrada")    AS "Ano Entrada",
       Ordem."DataSaida" 
  FROM      "vw_fOSOficina" Ordem
  LEFT JOIN "@AGRI_UNPF"    Fazenda ON Ordem."IdFazenda"     = Fazenda."Code" 
  LEFT JOIN "@AGRI_MTEM"    Motivo  ON Ordem."IdMotivo"      = Motivo."Code"
  LEFT JOIN "@AGRI_CLAM"    Classe  ON Ordem."IdClasse"      = Classe."Code" 
  LEFT JOIN "@AGRI_EQPM"    Equip   ON Ordem."IdEquipamento" = Equip."Code"
